user {
    int user_id PK
    varchar username "用户名"
    varcahr password "密码"
    varchar email "邮箱"
    date create_at "注册时间"
    int user_role "用户角色"
}

family {
    int family_id PK
    varchar family_name "基因家族"
}

gene_family {
    int gene_id FK
    int family_id FK
}

gene {
    int gene_id PK
    int user_id FK
    varchar gene_symbol "基因名"
    varcahr gene_acc "编号"
    varcahr gene_seq "序列"
    varcahr gene_msg "相关信息"
    int taxon_id FK
}

taxon {
    int taxon_id PK
    varcahr taxon_name "物种名"
}

annot {
    int annot_id PK
    int gene_id FK
    varcahr annot_acc "外库编号"
    int fdb_id FK
}

fdb {
    int fdb_id PK
    varchar fdb_name "数据库名"
}
